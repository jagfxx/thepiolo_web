# Nginx — THEPIOLO Portfolio
# Copiar en: /etc/nginx/sites-available/thepiolo-web
# Dominio: thepiolo.icu

# HTTP → HTTPS
server {
    listen 80;
    server_name thepiolo.icu www.thepiolo.icu;

    return 301 https://$server_name$request_uri;
}

# HTTPS
server {
    listen 443 ssl http2;
    server_name thepiolo.icu www.thepiolo.icu;

    ssl_certificate /etc/letsencrypt/live/thepiolo.icu/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/thepiolo.icu/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;

    access_log /var/log/nginx/thepiolo-web-access.log;
    error_log /var/log/nginx/thepiolo-web-error.log;

    client_max_body_size 10M;

    # Puerto 3001 — TSM sigue en 3000
    # Un solo location: los bloques separados para /_next/static sin
    # proxy_set_header Host provocaban 400 en chunks JS/CSS de Next.js.
    location / {
        proxy_pass http://localhost:3001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }
}
